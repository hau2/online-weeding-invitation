API Endpoints (all require admin auth via credentials:include):
- GET /admin/custom-themes -> CustomThemeListItem[]
- GET /admin/custom-themes/:slug -> CustomTheme (full config)
- POST /admin/custom-themes { name, baseTheme } -> CustomTheme (draft)
- PUT /admin/custom-themes/:id (FormData with config JSON + optional backgroundImage file) -> CustomTheme
- POST /admin/custom-themes/:id/publish -> { status: 'published' }
- POST /admin/custom-themes/:id/disable -> { status: 'disabled' }
- DELETE /admin/custom-themes/:id -> { deleted: true }

```typescript
async function handleSave(publish = false) {
  setSaving(true)
  const formData = new FormData()
  formData.append('config', JSON.stringify(config))
  formData.append('name', config.name)
  if (publish) formData.append('status', 'published')
  if (backgroundImageFile) formData.append('backgroundImage', backgroundImageFile)

  const { data, error } = await apiUpload(`/admin/custom-themes/${theme.id}`, formData, 'PUT')
  if (error) { toast.error(error); setSaving(false); return }
  toast.success(publish ? 'Da xuat ban giao dien!' : 'Da luu nhap')
  if (publish) {
    await apiFetch(`/admin/custom-themes/${theme.id}/publish`, { method: 'POST', credentials: 'include' })
  }
  setSaving(false)
}
```
